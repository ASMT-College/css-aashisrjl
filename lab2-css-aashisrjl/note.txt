1. Variables====================================
C:

#include <stdio.h>
int main() {
    int x = 5; // Declaration with initialization
    printf("Value of x: %d", x);
    return 0;
}

JavaScript:

let x = 5; // Declaration with initialization
console.log("Value of x:", x);
Difference:


C requires explicit variable types (int, float, char, etc.).
JavaScript uses let, const, or var (no explicit type required).


2. Data Types==========================================
Data Type	C	JavaScript
Numbers	int, float	Number
Strings	char[]	String
Boolean	_Bool	Boolean
Arrays	int arr[5]	[1, 2, 3]
Objects	N/A	{key: value}


3. Statements=============================================
C:

int x = 10;
if (x > 5) {
    printf("x is greater than 5");
}

JavaScript:

let x = 10;
if (x > 5) {
    console.log("x is greater than 5");
}
4. Operators
Type	C	JavaScript
Arithmetic	+, -, /	+, -, /
Logical	&&, `	
Comparison	==, !=	===, !==, >


5. Flow Control (If-Else)======================================
C:

#include <stdio.h>
int main() {
    int num = 10;
    if (num > 5) {
        printf("Greater than 5");
    } else {
        printf("Less or equal to 5");
    }
    return 0;
}

JavaScript:

let num = 10;
if (num > 5) {
    console.log("Greater than 5");
} else {
    console.log("Less or equal to 5");
}


6. Loops ==================================================
C:

#include <stdio.h>
int main() {
    for (int i = 0; i < 5; i++) {
        printf("%d ", i);
    }
    return 0;
}

JavaScript:

for (let i = 0; i < 5; i++) {
    console.log(i);
}


7. Functions===========================

C:
#include <stdio.h>
int add(int a, int b) {
    return a + b;
}
int main() {
    printf("Sum: %d", add(3, 4));
    return 0;
}

JavaScript:

function add(a, b) {
    return a + b;
}
console.log("Sum:", add(3, 4));
